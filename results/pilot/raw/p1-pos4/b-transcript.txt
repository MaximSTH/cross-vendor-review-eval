  return {
    workspaceRoot,
    job: enrichJob(selected, { maxProgressLines: options.maxProgressLines })
  };
}

export function resolveResultJob(cwd, reference) {
  const workspaceRoot = resolveWorkspaceRoot(cwd);
  const jobs = sortJobsNewestFirst(reference ? listJobs(workspaceRoot) : filterJobsForCurrentSession(listJobs(workspaceRoot)));
  const selected = matchJobReference(
    jobs,
    reference,
    (job) => job.status === "completed" || job.status === "failed" || job.status === "cancelled"
  );

  if (selected) {
    return { workspaceRoot, job: selected };
  }

  const active = matchJobReference(jobs, reference, (job) => job.status === "queued" || job.status === "running");
  if (active) {
    throw new Error(`Job ${active.id} is still ${active.status}. Check /codex:status and try again once it finishes.`);
  }

  if (reference) {
    throw new Error(`No finished job found for "${reference}". Run /codex:status to inspect active jobs.`);
  }

  throw new Error("No finished Codex jobs found for this repository yet.");
}

export function resolveCancelableJob(cwd, reference) {
  const workspaceRoot = resolveWorkspaceRoot(cwd);
  const jobs = sortJobsNewestFirst(listJobs(workspaceRoot));
  const activeJobs = jobs.filter((job) => job.status === "queued" || job.status === "running");

  if (reference) {
    const selected = matchJobReference(activeJobs, reference);
    if (!selected) {
      throw new Error(`No active job found for "${reference}".`);
    }
    return { workspaceRoot, job: selected };
  }

  if (activeJobs.length === 1) {
    return { workspaceRoot, job: activeJobs[0] };
  }
  if (activeJobs.length > 1) {
    throw new Error("Multiple Codex jobs are active. Pass a job id to /codex:cancel.");
  }

  throw new Error("No active Codex jobs to cancel.");
}

async function waitForSingleJobSnapshot(cwd, reference, options = {}) {
  const timeoutMs = Math.max(0, Number(options.timeoutMs) || DEFAULT_STATUS_WAIT_TIMEOUT_MS);
  const pollIntervalMs = Math.max(100, Number(options.pollIntervalMs) || DEFAULT_STATUS_POLL_INTERVAL_MS);
  const deadline = Date.now() + timeoutMs;
  let snapshot = buildSingleJobSnapshot(cwd, reference);

  while (isActiveJobStatus(snapshot.job.status) && Date.now() < deadline) {
    await sleep(Math.min(pollIntervalMs, Math.max(0, deadline - Date.now())));
    snapshot = buildSingleJobSnapshot(cwd, reference);
  }

  return {
    ...snapshot,
    waitTimedOut: isActiveJobStatus(snapshot.job.status),
    timeoutMs
  };
}

    sandbox: request.write ? "workspace-write" : "read-only",
    onProgress: request.onProgress,
    persistThread: true,
    threadName: resumeThreadId ? null : buildPersistentTaskThreadName(request.prompt || DEFAULT_CONTINUE_PROMPT)
  });

  const failureMessage = result.error?.message ?? result.stderr ?? "";

      rawOutput,
      failureMessage,

      title: taskMetadata.title,
      jobId: request.jobId ?? null,
      write: Boolean(request.write)

    status: result.status,
    threadId: result.threadId,
    rawOutput,

    reasoningSummary: result.reasoningSummary

  return {
    exitStatus: result.status,
    threadId: result.threadId,
    turnId: result.turnId,
    payload,
    rendered,
    summary: firstMeaningfulLine(rawOutput, firstMeaningfulLine(failureMessage, `${taskMetadata.title} finished.`)),

async function handleTaskWorker(argv) {

    throw new Error(`Stored job ${options["job-id"]} is missing its task request payload.`);

The diff scopes `--resume-last` and the resume-candidate lookup to the current Claude session (via the newly-exported `filterJobsForCurrentSession`/`getCurrentSessionId`), and gates the on-disk thread fallback behind `sessionId ? null : findLatestTaskThread(...)`. The three new tests confirm the intended session-isolation behavior, and job records are stamped with `sessionId` so same-session resume still works.

The change is largely coherent and well-tested. The one issue I can identify is a behavioral inconsistency introduced by disabling the disk fallback when a session is active:

- `task-resume-candidate` advertises any non-active task (including `failed`/`cancelled`) with a `threadId` as an available resume candidate (line 872-878).
- But `resolveLatestTrackedTaskThread` only accepts `status === "completed"` tracked tasks (line 323), and now, when a session id is set, returns `null` instead of falling back to `findLatestTaskThread` (line 328). Previously the disk fallback would have recovered a failed/cancelled task's persisted thread.

So in session mode, after a failed (but thread-persisted) task, the UI hint says a resumable task exists, yet `/codex:task --resume-last` errors with "No previous Codex task thread was found." Before this change that path resumed successfully via the disk fallback.
