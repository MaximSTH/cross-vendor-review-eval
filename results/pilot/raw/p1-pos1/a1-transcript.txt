// RFC 2046 §5.1.1:
//   delimiter        = "--" boundary [transport-padding] CRLF
//   close-delimiter  = "--" boundary "--" [transport-padding] CRLF
// `--${boundary}` must be followed only by whitespace/CRLF, or by `--`
// (close) followed by whitespace/CRLF. A line like `--abcXYZ` is *not* a
// delimiter even though it shares the prefix `--abc`.
function matchMimeBoundary(line, boundary) {
  if (!boundary) return null
  const prefix = `--${boundary}`
  if (!line.startsWith(prefix)) return null
  const tail = line.slice(prefix.length)
  if (tail.startsWith('--')) {
    return /^[ \t]*\r?\n?$/.test(tail.slice(2)) ? 'close' : null
  }
  return /^[ \t]*\r?\n?$/.test(tail) ? 'open' : null
}

class ChunkEmitter extends EventEmitter {
  constructor(buffer_size) {
    super()
    this.buffer_size = parseInt(buffer_size) || 64 * 1024
    this.buf = null
    this.pos = 0
    this.bufs = []

  }

  fill(input) {
    if (typeof input === 'string') input = Buffer.from(input)

    if (!this.buf) {
      this.bufs.push(input)
      this.bufs_size += input.length
      if (input.length + this.bufs_size > this.buffer_size) {
        this.buf = Buffer.alloc(this.buffer_size)
        const in_new = Buffer.concat(this.bufs, this.bufs_size)
        input = in_new
        this.bufs = []
        this.bufs_size = 0
      } else {
        return
      }
    }

    while (input.length > 0) {
      let remaining = this.buffer_size - this.pos
      if (remaining === 0) {
        this.emit('data', this.buf)
        this.buf = Buffer.alloc(this.buffer_size)
        this.pos = 0
        remaining = this.buffer_size
      }
      const to_write = remaining > input.length ? input.length : remaining
      input.copy(this.buf, this.pos, 0, to_write)
      this.pos += to_write
      input = input.slice(to_write)
    }
  }

  end(cb) {
    let emitted = false
    if (this.bufs_size > 0) {
      this.emit('data', Buffer.concat(this.bufs, this.bufs_size))
      emitted = true
    } else if (this.pos > 0) {
      this.emit('data', this.buf.slice(0, this.pos))
      emitted = true
    }
    this.buf = null
    this.pos = 0
    this.bufs = []
    this.bufs_size = 0
    if (typeof cb === 'function') cb()
    return emitted
  }

I reviewed my own change with fresh eyes and found a significant regression I introduced.

## Findings

**1. The streaming header-shape heuristic truncates well-formed header blocks (`transaction.js:151`)**

This is the serious one. The old code only applied the "does this line look like a header?" heuristic in `end_data`, and only for messages that never sent a blank-line separator. I moved it into `add_data` so it runs against *every* line of *every* message. A single malformed line now ends the header block, and every legitimate header after it is silently reclassified as body.

Concretely, for `Subject : hi` (space before the colon — common in the wild):

```
Received: from x        →  header
Subject : hi            →  [^\s:]*: fails; ' ' is \s so it can't reach the colon
To: a@b.c               →  BODY
                        →  BODY
body                    →  BODY
```

`transaction.header.get('to')` returns `''`. Old code pushed all three lines into `header_lines`, and `header.parse()` logged "Header did not look right" for the bad one while still parsing `Received` and `To` correctly. Same failure for an mbox `From ` line at the top, or any header line missing its colon.

Downstream impact is broad: DKIM/DMARC/SPF alignment, subject rewriting, and the mail-loop guard at `connection.js:1573` (`header.get_all('received').length > max_received`) all read from `transaction.header`. A message crafted with one odd line early in the block hides every subsequent `Received:` from that check.

I don't think this is fixable by tightening the regex — the old design deliberately *deferred* the decision to `end_data` precisely because you can't know a blank line is coming. Restricting the streaming split to messages that reach `end_data` without a separator would preserve the fix for the reported bug without this regression.

**2. `end_data` now acts on completely empty DATA (`transaction.js:190`)**

I dropped `&& this.header_lines.length` from the guard. For `DATA\r\n.\r\n`, the new code synthesizes a `\r\n` into the message stream (a 0-byte message becomes 2 bytes) and, when `parse_body` is set, calls `body.parse_end()` on a `Body` that was never fed — which runs `_empty_filter()`, invoking banner and body-filter callbacks against an empty message. Old code did neither.

## Things I checked and cleared

- The leading-dot handling changed (header lines are now stored dot-stuffed rather than stripped, since I introduced `hline` instead of reassigning `line`). Both round-trip correctly through `LineTransformer`, and the new behavior is actually more correct under `dot_stuffed: false`.
- The shared module-level `hb_sep` buffer is retained by reference in `ChunkEmitter.bufs`, but it's always copied out via `Buffer.concat`/`input.copy` and nothing mutates it.
- `if (this.parse_body && this.body)` is strictly safer than the old `if (this.header_pos && this.parse_body)`, which would TypeError when `parse_body` was enabled after headers were parsed.
- Existing fixtures (`test/fixtures/util_hmailitem.js`, the multipart tests) are all well-formed and unaffected.

Worth noting about my own tests: the `normal message` case passes both before and after the change, so it isn't a regression test — and none of the five cases would have caught finding #1.
