27	function _parse_rfc2231(params, str) {
28	  // Fast-path: an RFC 2231 parameter always contains both `*` and `=`; skip the
29	  // scan for the common header (Subject, From, ...) that has neither.
30	  if (str.indexOf('*') === -1 || str.indexOf('=') === -1) return

36	    if (lang_match) {
37	      if (attribute === params.cur_key && lang_match[2] !== params.cur_lang) {
38	        continue

40	      params.cur_enc = lang_match[1]
41	      params.cur_lang = lang_match[2]

44	      params.cur_lang = ''
45	      params.cur_enc = ''
46	    }

function stream_line_reader(stream, transaction, cb) {
    let current_data = ''
    function process_data(data) {
        current_data += data.toString()
        let results
        while ((results = utils.line_regexp.exec(current_data))) {
            const this_line = results[1]
            current_data = current_data.slice(this_line.length)

                return
            }
            transaction.add_data(Buffer.from(this_line))
        }
    }

    function process_end() {
        if (current_data.length) {
            transaction.add_data(Buffer.from(current_data))
        }
        current_data = ''
        transaction.message_stream.add_line_end()
        cb()
    }

    stream.once('end', process_end)
    stream.once('error', cb)

const MAX_HEADER_SIZE = 1024 * 1024 // 1MB

function indexOfLF(buf) {
  for (let i = 0; i < buf.length; i++) {
    if (buf[i] === 0x0a) return i
  }
  return -1
}

// Strips everything from the start of a stored message up to and including
// the blank EOH line (\r\n or \n), then passes body bytes through unchanged.
class HeaderSkipper extends Transform {
  #remainder = Buffer.alloc(0)
  #eohFound = false

  _transform(chunk, enc, cb) {
    if (this.#eohFound) {
      this.push(chunk)
      return cb()
    }

    if (this.#remainder.length + chunk.length > MAX_HEADER_SIZE) {
      return cb(new Error('Header size limit exceeded'))
    }

    let buf = Buffer.concat([this.#remainder, chunk])
    let offset
    while ((offset = indexOfLF(buf)) !== -1) {
      const line = buf.slice(0, offset + 1)
      buf = buf.slice(offset + 1)
      const isEoh =
        (line.length === 2 && line[0] === 0x0d && line[1] === 0x0a) ||
        (line.length === 1 && line[0] === 0x0a)
      if (isEoh) {
        this.#eohFound = true
        if (buf.length > 0) this.push(buf)
        this.#remainder = Buffer.alloc(0)
        return cb()
      }
    }
    this.#remainder = buf
    cb()
  }

  _flush(cb) {
    // Any remainder after EOF with no EOH found is discarded (headers-only msg edge case)
    if (this.#remainder.length > 0 && this.#eohFound) this.push(this.#remainder)
    cb()
  }
}

## Root cause

`transaction.js` overloaded `header_pos` — a *line count* — as the boolean "have we passed the header/body separator?" flag. There are two ways it lands on `0` even though headers were already parsed:

1. **`add_data:136`** — `header_pos = this.header_lines.length`. A message whose DATA opens with a blank line (no headers of its own) parses to zero header lines, so `header_pos` stays `0` while `found_hb_sep` is `true`.
2. **`end_data:172`** — the busted-email fallback assigned the loop *index* of the last header line, not a count. One header line → `header_pos = 0`. And when the very first line wasn't header-shaped, `splice(header_pos + 1)` kept that body line as a header.

Once `header_pos` is stuck at `0`, `add_data` keeps re-entering the header branch, so **body lines get accumulated as headers** and the next blank line in the body re-triggers `header.parse()`. `add_header` also stops calling `reset_headers`. The message that gets serialized has a bogus header block made of body text, with the plugin-appended headers (`X-Spam-*`, `DKIM-Signature`) attached to it — which is what your capture shows, while `add_leading_header` (`Received`) survived at the top.

A second defect compounds it: for a message with no blank line at all, the stored byte stream has no EOH marker. `HeaderSkipper` strips raw bytes up to the *first blank line it finds*, which is somewhere inside the body — so the leading part of the body was silently eaten, or discarded entirely if the body had no blank line.

## Changes

- `found_hb_sep` is now the sole "headers are done" flag; `header_pos` goes back to being just a line count.
- Header/body splitting moved into `add_data`: a line that is neither header- nor continuation-shaped ends the header block right there, and a `\r\n` separator is written into the message stream. The stored message now always has a real EOH, so `HeaderSkipper` cuts in the right place and the body survives.
- `end_data`'s fallback shrank to the only case that can still reach it (every line looked like a header → no body).
- `add_header` / `add_leading_header` / `remove_header` guard on `found_hb_sep`.

Body lines from busted messages now also run through the normal `parse_body` path, so banners and body filters apply to them — previously `end_data` fed them to `body.parse_more()` and threw the result away.
